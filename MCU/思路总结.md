#### 相关函数
```

u8 str[] = "Hello World \r\n";

// 发送一个字符串函数
无返回值
void usart1_send_str(u8 *str)
{
    while(*str != '\0')
    {
        usart1_send_byte(*str);
        str++;
    }
}

u8 str[10];
// 接收一个字符串函数
// 无返回值
void usart1_rec_str(u8 *str)
{

    while(1)
    {
        *str = usart1_rec_byte();
        if (*str == '#')
        {
            break;
        }
        str++;
    }

    *str = '\0';
}

```

        // 清除空闲标志位
        // 执行事件
    }
}

// 中断服务函数
